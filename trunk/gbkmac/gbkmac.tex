\def\cjkchar#1#2#3{%for char whole in one subfont plane
  \chardef\cjkgap 127%
  \tempcnta `#3%
  \ifnum `#3 > \cjkgap
     \advance\tempcnta -1%
  \fi
  \advance\tempcnta #2%
  \edef\cjkplane{#1}% 
\chifont{\cjkplane}\savefont{\the\font}\the\font\cjkfont\cglue\nobreak\char\tempcnta\ccmaphook\relax\restorefont%
}
\def\cjkcharx#1#2#3#4{%for char may cross two subfont plane
  \chardef\cjkgap 127%
  \tempcnta `#4%
  \tempcntb 256%
%
  \ifnum `#4 > \cjkgap
    \advance\tempcnta -1%
    \advance\tempcntb 1%
  \fi
  \advance\tempcnta #3%
  \advance\tempcntb -#3%
  \chardef\temp \tempcntb
  \ifnum `#4 < \temp
     \edef\cjkplane{#1}% 
  \else
     \advance\tempcnta -256%
     \edef\cjkplane{#2}%
  \fi%
\chifont{\cjkplane}\savefont{\the\font}\the\font\cjkfont\cglue\nobreak\char\tempcnta\ccmaphook\relax\restorefont%
}
