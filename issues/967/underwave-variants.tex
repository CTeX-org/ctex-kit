\newcommand\edge{\color{red}\vrule width .25pt height 1.1em depth .45em}
\newcommand\marked[1]{%
  \makebox[0pt][r]{\edge}%
  #1%
  \makebox[0pt][l]{\edge}}
\newcommand\normal{\marked{\CJKunderwave{国国国国国国}}}
\newcommand\subtract{\marked{\CJKunderwave-{国国国}\CJKunderwave-{国国国}}}
\newcommand\row[2]{%
  \makebox[32mm][l]{\ttfamily\small #1}%
  \hskip#2\relax \normal\qquad\subtract\par\medskip}
\newcommand\variant[2]{%
  \section*{#1}
  \begingroup
  #2
  \fontsize{26}{38}\selectfont
  \row{offset 0pt}{0pt}
  \row{offset 1pt}{1pt}
  \row{offset 3pt}{3pt}
  \row{offset 5pt}{5pt}
  \endgroup
  \clearpage}

\begin{document}
