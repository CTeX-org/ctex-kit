\newcommand\sample{国国国国}
\newcommand\offsetrow[2]{%
  \noindent\makebox[34mm][l]{\ttfamily\scriptsize #1}%
  \hskip#2\relax
  \CJKunderdblline{\sample}\quad
  \CJKunderwave{\sample}\quad
  \CJKsout{\sample}\quad
  \CJKxout{\sample}\quad
  \CJKunderanyline{.2em}{\rule{.2em}{.8pt}}{\sample}\par
}
\newcommand\subtractrow[1]{%
  \noindent\makebox[34mm][l]{\ttfamily\scriptsize #1}%
  \CJKunderdblline-{国国}\CJKunderdblline-{国国}\quad
  \CJKunderwave-{国国}\CJKunderwave-{国国}\quad
  \CJKsout-{国国}\CJKsout-{国国}\quad
  \CJKunderanyline-{.2em}{\rule{.2em}{.8pt}}{国国}%
  \CJKunderanyline-{.2em}{\rule{.2em}{.8pt}}{国国}\par
}
\newcommand\punctrow[1]{%
  \noindent\makebox[34mm][l]{\ttfamily\scriptsize #1}%
  \CJKunderdblline{国国，国国。}\quad
  \CJKunderwave{国国，国国。}\quad
  \CJKsout{国国，国国。}\quad
  \CJKxout{国国，国国。}\par
}
\newcommand\linebreakrow[1]{%
  \noindent\makebox[34mm][l]{\ttfamily\scriptsize #1}%
  \parbox[t]{70mm}{\CJKunderwave{天地玄黄宇宙洪荒日月盈昃辰宿列张寒来暑往秋收冬藏}}\quad
  \parbox[t]{70mm}{\CJKxout{天地玄黄宇宙洪荒日月盈昃辰宿列张寒来暑往秋收冬藏}}\par
}
\newcommand\leadermatrix[2]{%
  \section*{#1}
  \begingroup
  \let\ULleaders#2
  \offsetrow{offset 0pt}{0pt}
  \offsetrow{offset 1pt}{1pt}
  \offsetrow{offset 3pt}{3pt}
  \offsetrow{offset 5pt}{5pt}
  \medskip
  \subtractrow{subtract}
  \medskip
  \punctrow{punctuation}
  \medskip
  \linebreakrow{line breaks}
  \endgroup
  \clearpage
}
