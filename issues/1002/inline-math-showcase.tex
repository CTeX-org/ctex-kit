\newcommand\InlineMathWrapper[1]{#1}

\makeatletter
\AtBeginDocument{%
  \expandafter\gdef\csname r@inline-math-target\endcsname
    {{1}{1}{title}{inline.math.target}{}}%
}
\makeatother

\newcommand\InlineMathPage[2]{%
  \CommandPage{#1}{%
    \noalign{\xeCJKsetup{xCJKecglue=false}}%
    \Matrix{xCJKecglue=false：中文—公式—中文}
      {中}{#2}{$x$}{文}%
    \Matrix{xCJKecglue=false：西文—公式—西文}
      {A}{#2}{$x$}{B}%
    \noalign{\xeCJKsetup{xCJKecglue=true}}%
    \Matrix{xCJKecglue=true：中文—公式—中文}
      {中}{#2}{$x$}{文}%
    \Matrix{xCJKecglue=true：西文—公式—西文}
      {A}{#2}{$x$}{B}%
  }%
}
