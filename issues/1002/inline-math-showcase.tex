\newcommand\InlineMathVariant[4]{%
  \begingroup
    \xeCJKsetup{xCJKecglue=#1}%
    \MatrixReset\global\setbox\MatrixCandidateContext=\hbox{#3}%
    \MatrixReset\global\setbox\MatrixOracleContext=\hbox{#4}%
  \endgroup
  \global\MatrixDelta=\dimexpr
    \wd\MatrixCandidateContext-\wd\MatrixOracleContext-
    \MatrixIntrinsicDelta\relax
  \global\MatrixAbsDelta=\MatrixDelta
  \ifdim\MatrixAbsDelta<0pt \global\MatrixAbsDelta=-\MatrixAbsDelta\fi
  \texttt{#2}&
  \fbox{\copy\MatrixOracleContext}&
  \fbox{\copy\MatrixCandidateContext}&
  \ifdim\MatrixAbsDelta<0.01pt
    \textcolor{teal!70!black}{same}%
  \else
    \textcolor{red!80!black}{\the\MatrixDelta}%
  \fi
  \tabularnewline
}

% #1 xCJKecglue，#2 标题，#3 左侧文字，#4 候选，#5 直接输入，#6 右侧文字。
\newcommand\InlineMathMatrix[6]{%
  \multicolumn{4}{@{}l}{%
    \begingroup
      \xeCJKsetup{xCJKecglue=#1}%
      \MatrixReset\global\setbox\MatrixCandidate=\hbox{#4}%
      \MatrixReset\global\setbox\MatrixOracle=\hbox{#5}%
    \endgroup
    \global\MatrixIntrinsicDelta=\dimexpr
      \wd\MatrixCandidate-\wd\MatrixOracle\relax
    \rule{0pt}{14pt}\bfseries xCJKecglue=#1：#2}\tabularnewline
  \InlineMathVariant{#1}{00}{#3#4#6}{#3#5#6}%
  \InlineMathVariant{#1}{10}{#3 #4#6}{#3 #5#6}%
  \InlineMathVariant{#1}{01}{#3#4 #6}{#3#5 #6}%
  \InlineMathVariant{#1}{11}{#3 #4 #6}{#3 #5 #6}%
}

\newcommand\InlineMathPage[2]{%
  \CommandPage{#1}{%
    \InlineMathMatrix{false}{中文—公式—中文}
      {中}{#2}{$x$}{文}%
    \InlineMathMatrix{false}{西文—公式—西文}
      {A}{#2}{$x$}{B}%
    \InlineMathMatrix{true}{中文—公式—中文}
      {中}{#2}{$x$}{文}%
    \InlineMathMatrix{true}{西文—公式—西文}
      {A}{#2}{$x$}{B}%
  }%
}
