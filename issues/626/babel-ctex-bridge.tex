% Issue #626: a public-key bridge between Babel language selection and CTeX.
\documentclass[fontset=fandol]{ctexbook}
\usepackage[paperwidth=100mm,paperheight=100mm,margin=9mm]{geometry}
\usepackage[english,chinese,provide=*]{babel}

% Babel selects the language; these hooks set only CTeX's public keys.
\newcommand\CTEXBabelChinese{%
  \ctexset{
    contentsname = 目录,
    listfigurename = 插图,
    listtablename = 表格,
    figurename = 图,
    tablename = 表,
    abstractname = 摘要,
    indexname = 索引,
    bibname = 参考文献,
    appendixname = 附录,
    proofname = 证明,
    part/name = {第,部分},
    part/number = \chinese{part},
    chapter/name = {第,章},
    chapter/number = \chinese{chapter},
  }%
}
\newcommand\CTEXBabelEnglish{%
  \ctexset{
    contentsname = Contents,
    listfigurename = List of Figures,
    listtablename = List of Tables,
    figurename = Figure,
    tablename = Table,
    abstractname = Abstract,
    indexname = Index,
    bibname = Bibliography,
    appendixname = Appendix,
    proofname = Proof,
    part/name = {Part~,},
    part/number = \thepart,
    chapter/name = {Chapter~,},
    chapter/number = \thechapter,
  }%
}
\addto\extraschinese{\CTEXBabelChinese}
\addto\extrasenglish{\CTEXBabelEnglish}

\begin{document}
\selectlanguage{chinese}
\part{中文部分}
\chapter{中文章节}
\begin{figure}[h]
  \centering\rule{20mm}{10mm}
  \caption{中文图题}
\end{figure}

\selectlanguage{english}
\part{English Part}
\chapter{English Chapter}
\begin{figure}[h]
  \centering\rule{20mm}{10mm}
  \caption{English figure caption}
\end{figure}
\end{document}
