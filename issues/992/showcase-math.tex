% #1 命令包裹后的内容，#2 直接输入，#3 源码空格，#4 组名，#5 行名。
\newcommand\InlinePrepareVariant[5]{%
  \MatrixReset\global\setbox\MatrixCandidateContext=\hbox{#1}%
  \MatrixReset\global\setbox\MatrixOracleContext=\hbox{#2}%
  \global\MatrixDelta=\dimexpr
    \wd\MatrixCandidateContext-\wd\MatrixOracleContext-
    \MatrixIntrinsicDelta\relax
  \global\MatrixAbsDelta=\MatrixDelta
  \ifdim\MatrixAbsDelta<0pt
    \global\MatrixAbsDelta=-\MatrixAbsDelta
  \fi
  \ifdim\MatrixAbsDelta<0.01pt
    \typeout{SHOWCASE|\InlineMathMode|\the\InlineCommandNumber/#4|#3|PASS|%
      delta=\the\MatrixDelta}%
  \else
    \typeout{SHOWCASE|\InlineMathMode|\the\InlineCommandNumber/#4|#3|FAIL|%
      delta=\the\MatrixDelta}%
  \fi
  \global\setbox\InlineRowBox{#4}{#5}=\hbox{%
    \makebox[10mm][c]{\texttt{#3}}\hspace{2\tabcolsep}%
    \makebox[54mm][c]{\fbox{\copy\MatrixOracleContext}}%
    \hspace{2\tabcolsep}%
    \makebox[54mm][c]{\fbox{\copy\MatrixCandidateContext}}%
    \hspace{2\tabcolsep}%
    \makebox[20mm][c]{%
      \ifdim\MatrixAbsDelta<0.01pt
        \textcolor{teal!70!black}{same}%
      \else
        \textcolor{red!80!black}{\the\MatrixDelta}%
      \fi}}%
}

% #1 组名，#2 左侧文字，#3 命令包裹后的内容，#4 直接输入，#5 右侧文字。
\newcommand\InlinePrepareMatrix[5]{%
  \MatrixReset\global\setbox\MatrixCandidate=\hbox{#3}%
  \MatrixReset\global\setbox\MatrixOracle=\hbox{#4}%
  \global\MatrixIntrinsicDelta=\dimexpr
    \wd\MatrixCandidate-\wd\MatrixOracle\relax
  \InlinePrepareVariant{#2#3#5}{#2#4#5}{00}{#1}{Zero}%
  \InlinePrepareVariant{#2 #3#5}{#2 #4#5}{10}{#1}{Left}%
  \InlinePrepareVariant{#2#3 #5}{#2#4 #5}{01}{#1}{Right}%
  \InlinePrepareVariant{#2 #3 #5}{#2 #4 #5}{11}{#1}{Both}%
}

\newcommand\InlineDisplayMatrix[2]{%
  \multicolumn{4}{@{}l}{%
    \rule{0pt}{14pt}\bfseries xCJKecglue=\InlineMathMode：#2}%
    \tabularnewline
  \multicolumn{4}{@{}l}{\copy\InlineRowBox{#1}{Zero}}\tabularnewline
  \multicolumn{4}{@{}l}{\copy\InlineRowBox{#1}{Left}}\tabularnewline
  \multicolumn{4}{@{}l}{\copy\InlineRowBox{#1}{Right}}\tabularnewline
  \multicolumn{4}{@{}l}{\copy\InlineRowBox{#1}{Both}}\tabularnewline
}

\newcommand\InlineMathPage[2]{%
  \advance\InlineCommandNumber by 1
  \InlinePrepareMatrix{CMC}{中}{#2}{$x$}{文}%
  \InlinePrepareMatrix{CMW}{中}{#2}{$x$}{B}%
  \InlinePrepareMatrix{WMW}{A}{#2}{$x$}{B}%
  \InlinePrepareMatrix{WMC}{A}{#2}{$x$}{文}%
  \CommandPage{#1}{%
    \InlineDisplayMatrix{CMC}{中文—公式—中文}%
    \InlineDisplayMatrix{CMW}{中文—公式—西文}%
    \InlineDisplayMatrix{WMW}{西文—公式—西文}%
    \InlineDisplayMatrix{WMC}{西文—公式—中文}%
  }%
}
