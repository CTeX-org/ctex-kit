% Shared rendering harness for per-command showcase pages.
% Same \Matrix call shape and \MatrixReset isolation as the
% command-boundary-*-matrix.tex audit drivers, but each variant renders
% an fbox comparison row instead of a MATRIX| log line.
%
% Distinguishing glue values are applied by every wrapper so that
% missing glue cannot be hidden by default natural widths.
\usepackage{array}
\usepackage{xcolor}

\pagestyle{empty}
\setlength{\parindent}{0pt}
\setlength{\fboxsep}{2pt}
\renewcommand\arraystretch{1.35}

\AtBeginDocument{%
  \xeCJKsetup{
    CJKecglue={\hskip 5pt},
    CJKglue={\hskip 1pt}
  }%
}

\newbox\MatrixCandidate
\newbox\MatrixOracle
\newbox\MatrixCandidateContext
\newbox\MatrixOracleContext
\newdimen\MatrixIntrinsicDelta
\newdimen\MatrixDelta
\newdimen\MatrixAbsDelta

\ExplSyntaxOn
\cs_new_protected:Npn \MatrixReset
  {
    \tl_gclear:N \g__xeCJK_last_node_tl
    \bool_gset_false:N \g__xeCJK_glue_check_pending_bool
    \bool_gset_false:N \g__xeCJK_reset_color_pending_bool
  }
\ExplSyntaxOff

\newcommand\MatrixVariant[3]{%
  \MatrixReset\global\setbox\MatrixCandidateContext=\hbox{#2}%
  \MatrixReset\global\setbox\MatrixOracleContext=\hbox{#3}%
  \global\MatrixDelta=\dimexpr
    \wd\MatrixCandidateContext-\wd\MatrixOracleContext-
    \MatrixIntrinsicDelta\relax
  \global\MatrixAbsDelta=\MatrixDelta
  \ifdim\MatrixAbsDelta<0pt \global\MatrixAbsDelta=-\MatrixAbsDelta\fi
  \texttt{#1}&
  \fbox{\copy\MatrixOracleContext}&
  \fbox{\copy\MatrixCandidateContext}&
  \ifdim\MatrixAbsDelta<0.01pt
    \textcolor{teal!70!black}{same}%
  \else
    \textcolor{red!80!black}{\the\MatrixDelta}%
  \fi
  \tabularnewline
}

% #1 group heading, #2 pre, #3 candidate, #4 oracle, #5 post
% Intrinsic-width measurement lives inside the \multicolumn cell: in a
% tabular a cell that uses \multicolumn may not have anything before it,
% and the box registers are set globally so the following variant rows
% still see them.
\newcommand\Matrix[5]{%
  \multicolumn{4}{@{}l}{%
    \MatrixReset\global\setbox\MatrixCandidate=\hbox{#3}%
    \MatrixReset\global\setbox\MatrixOracle=\hbox{#4}%
    \global\MatrixIntrinsicDelta=\dimexpr
      \wd\MatrixCandidate-\wd\MatrixOracle\relax
    \rule{0pt}{14pt}\bfseries #1}\tabularnewline
  \MatrixVariant{00}{#2#3#5}{#2#4#5}%
  \MatrixVariant{10}{#2 #3#5}{#2 #4#5}%
  \MatrixVariant{01}{#2#3 #5}{#2#4 #5}%
  \MatrixVariant{11}{#2 #3 #5}{#2 #4 #5}%
}

\newcolumntype{D}{>{\centering\arraybackslash}p{10mm}}
\newcolumntype{O}{>{\centering\arraybackslash}p{54mm}}
\newcolumntype{W}{>{\centering\arraybackslash}p{20mm}}

% #1 command display name, #2 table body
\newcommand\CommandPage[2]{%
  {\large\bfseries \texttt{\detokenize{#1}}}\quad
  {\footnotesize CJKecglue=5pt, CJKglue=1pt; framed boxes show natural
   width; $\Delta$ = candidate $-$ oracle after removing intrinsic width}\par
  \smallskip
  \begin{tabular}{@{}D O O W@{}}
  \bfseries space&
  \bfseries direct-input oracle&
  \bfseries command candidate&
  \bfseries $\Delta$\tabularnewline
  \hline
  #2%
  \end{tabular}%
  \clearpage
}
