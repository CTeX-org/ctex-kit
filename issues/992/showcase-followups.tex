% Issue #996 compares two boxes built from identical input.  Keep the
% state between the two assignments: resetting it would hide the bug that
% the example is meant to catch.
\newbox\CrossboxFirst
\newbox\CrossboxSecond
\newbox\CrossboxProtected
\newdimen\CrossboxDelta
\newdimen\CrossboxAbsDelta

\newcommand\CrossboxRow[3]{%
  \global\CrossboxDelta=\dimexpr\wd#3-\wd#2\relax
  \global\CrossboxAbsDelta=\CrossboxDelta
  \ifdim\CrossboxAbsDelta<0pt
    \global\CrossboxAbsDelta=-\CrossboxAbsDelta
  \fi
  \texttt{#1}&
  \fbox{\copy#2}&
  \fbox{\copy#3}&
  \ifdim\CrossboxAbsDelta<0.01pt
    \textcolor{teal!70!black}{same}%
  \else
    \textcolor{red!80!black}{\the\CrossboxDelta}%
  \fi
  \tabularnewline
}
