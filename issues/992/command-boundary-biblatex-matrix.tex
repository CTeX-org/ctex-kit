  {
    \tl_gclear:N \g__xeCJK_last_node_tl
    \bool_gset_false:N \g__xeCJK_glue_check_pending_bool
    \bool_gset_false:N \g__xeCJK_reset_color_pending_bool
  }
\ExplSyntaxOff

\newcommand\MatrixVariant[3]{%
  \MatrixReset\setbox\MatrixCandidateContext=\hbox{#2}%
  \MatrixReset\setbox\MatrixOracleContext=\hbox{#3}%
  \MatrixDelta=\dimexpr
    \wd\MatrixCandidateContext-\wd\MatrixOracleContext-
    \MatrixIntrinsicDelta\relax
  \ifdim\MatrixDelta<0pt \MatrixDelta=-\MatrixDelta\fi
  \ifdim\MatrixDelta<0.01pt
    \typeout{MATRIX|\MatrixLabel|#1|PASS|delta=\the\MatrixDelta}%
  \else
    \typeout{MATRIX|\MatrixLabel|#1|FAIL|delta=\the\MatrixDelta}%
  \fi
}

\newcommand\Matrix[5]{%
  \def\MatrixLabel{#1}%
  \MatrixReset\setbox\MatrixCandidate=\hbox{#3}%
  \MatrixReset\setbox\MatrixOracle=\hbox{#4}%
  \MatrixIntrinsicDelta=\dimexpr
    \wd\MatrixCandidate-\wd\MatrixOracle\relax
  \MatrixVariant{00}{#2#3#5}{#2#4#5}%
  \MatrixVariant{10}{#2 #3#5}{#2 #4#5}%
  \MatrixVariant{01}{#2#3 #5}{#2#4 #5}%
  \MatrixVariant{11}{#2 #3 #5}{#2 #4 #5}%
}

% Simulate biblatex's \blx@pagetracker exactly as xeCJK/testfiles/
% biblatex-ecglue01.lvt does: a \protected@write whatsit with the same
% shape biblatex emits at \blx@bibitem entry. Defining it in the preamble
% makes xeCJK's begindocument/before hook install the #931 patch.
\makeatletter
\newwrite\bibaux
\immediate\openout\bibaux=\jobname-bibaux.aux
\def\blx@pagetracker{%
  \protected@write\bibaux{}{\string\abx@aux@page{1}{\the\c@page}}%
}
\newcommand\PageTracker{\blx@pagetracker}
\makeatother

\begin{document}

% The write whatsit fires before the visible entry text, matching the
% real biblatex call site (\blx@bibitem). The whatsit itself has no
% visible output, so the oracle is the bare entry text.
%
% Only the before-content position is expressible in all four
% source-space combinations: a control word gobbles following spaces,
% so "Alpha\PageTracker<space>文" would silently degenerate to the
% no-space cell and misreport 01/11.
\Matrix{pagetracker/W/CWC}{中}{\PageTracker Alpha}{Alpha}{文}
\Matrix{pagetracker/C/WCW}{A}{\PageTracker 中文}{中文}{B}
\Matrix{pagetracker/C/CCC}{中}{\PageTracker 中文}{中文}{文}

\end{document}
