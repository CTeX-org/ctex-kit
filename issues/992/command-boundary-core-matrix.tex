% Compare one command-wrapped expression with its visible-text oracle.
% The standalone width difference removes the command's intrinsic width;
% the remaining delta is entirely due to the surrounding boundaries.
\newcommand\MatrixVariant[3]{%
  \MatrixReset
  \setbox\MatrixCandidateContext=\hbox{#2}%
  \MatrixReset
  \setbox\MatrixOracleContext=\hbox{#3}%
  \MatrixDelta=\dimexpr
    \wd\MatrixCandidateContext-\wd\MatrixOracleContext-
    \MatrixIntrinsicDelta\relax
  \ifdim\MatrixDelta<0pt
    \MatrixDelta=-\MatrixDelta
  \fi
  \ifdim\MatrixDelta<0.01pt
    \typeout{MATRIX|\MatrixLabel|#1|PASS|delta=\the\MatrixDelta}%
  \else
    \typeout{MATRIX|\MatrixLabel|#1|FAIL|delta=\the\MatrixDelta}%
  \fi
}

% Four literal-source-space combinations: 00, 10, 01, 11.
\newcommand\Matrix[5]{%
  \def\MatrixLabel{#1}%
  \MatrixReset
  \setbox\MatrixCandidate=\hbox{#3}%
  \MatrixReset
  \setbox\MatrixOracle=\hbox{#4}%
  \MatrixIntrinsicDelta=\dimexpr
    \wd\MatrixCandidate-\wd\MatrixOracle\relax
  \MatrixVariant{00}{#2#3#5}{#2#4#5}%
  \MatrixVariant{10}{#2 #3#5}{#2 #4#5}%
  \MatrixVariant{01}{#2#3 #5}{#2#4 #5}%
  \MatrixVariant{11}{#2 #3 #5}{#2 #4 #5}%
}

\newcommand\Baseline[5]{%
  \MatrixReset\setbox0=\hbox{#2#3#4}%
  \MatrixReset\setbox1=\hbox{#2 #3#4}%
  \MatrixReset\setbox2=\hbox{#2#3 #4}%
  \MatrixReset\setbox3=\hbox{#2 #3 #4}%
  \typeout{BASELINE|#1|00=\the\wd0|10=\the\wd1|01=\the\wd2|11=\the\wd3}%
}
