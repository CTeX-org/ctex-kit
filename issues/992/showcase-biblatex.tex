% Stub biblatex's \blx@pagetracker with the exact whatsit shape it emits
% (same stub as xeCJK/testfiles/biblatex-ecglue01.lvt); defining it in
% the preamble makes xeCJK install its #931 patch at begindocument.
\makeatletter
\newwrite\bibaux
\immediate\openout\bibaux=\jobname-bibaux.aux
\def\blx@pagetracker{%
  \protected@write\bibaux{}{\string\abx@aux@page{1}{\the\c@page}}%
}
\newcommand\PageTracker{\blx@pagetracker}
\makeatother

\begin{document}

\CommandPage{\blx@pagetracker (biblatex write whatsit, issue 931)}{%
  \Matrix{Western entry: 中\string\PageTracker Alpha文}{中}{\PageTracker Alpha}{Alpha}{文}
  \Matrix{CJK entry: A\string\PageTracker 中文B}{A}{\PageTracker 中文}{中文}{B}
  \Matrix{CJK entry: 中\string\PageTracker 中文文}{中}{\PageTracker 中文}{中文}{文}
}

\end{document}
