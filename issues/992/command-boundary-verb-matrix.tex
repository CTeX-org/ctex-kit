\ExplSyntaxOn
\cs_new_protected:Npn \MatrixReset
  {
    \tl_gclear:N \g__xeCJK_last_node_tl
    \bool_gset_false:N \g__xeCJK_glue_check_pending_bool
    \bool_if_exist:NT \g__xeCJK_reset_color_pending_bool
      { \bool_gset_false:N \g__xeCJK_reset_color_pending_bool }
  }
\ExplSyntaxOff

\newcommand\Report[2]{%
  \BoundaryDelta=\dimexpr
    \wd\CandContext-\wd\RawContext-\IntrinsicDelta\relax
  \ifdim\BoundaryDelta<0pt \BoundaryDelta=-\BoundaryDelta\fi
  \ifdim\BoundaryDelta<0.01pt
    \typeout{MATRIX|#1|#2|PASS|delta=\the\BoundaryDelta}%
  \else
    \typeout{MATRIX|#1|#2|FAIL|delta=\the\BoundaryDelta}%
  \fi
}
