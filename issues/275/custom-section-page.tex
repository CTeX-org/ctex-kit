\documentclass[fontset=fandol]{ctexbeamer}

\ctexset
  {
    section =
      {
        name = {第,节},
        number = \chinese{section}
      }
  }

\setbeamertemplate{section page}
  {
    \typeout{ISSUE275-CTEXTHESECTION: \CTEXthesection}
    \ifcsname CTEXifname\endcsname
      \typeout{ISSUE275-IFNAME: defined}
    \else
      \typeout{ISSUE275-IFNAME: undefined}
    \fi
    \ifodd\csname CTEX@section@numbering\endcsname
      \typeout{ISSUE275-PRIVATE-NUMBERING: true}
    \else
      \typeout{ISSUE275-PRIVATE-NUMBERING: false}
    \fi
    \CTEXthesection\quad\insertsection
  }
\AtBeginSection{\begin{frame}\sectionpage\end{frame}}

\begin{document}
\section{编号开启}
\ctexset{section/numbering=false}
\section{编号关闭}
\end{document}
