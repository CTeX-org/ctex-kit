\documentclass[fontset=fandol,aspectratio=169]{ctexbeamer}

\ExplSyntaxOn
\NewExpandableDocumentCommand \CTEXheadingnumber { m }
  { \use:c { CTEX@the#1 } }
\NewExpandableDocumentCommand \CTEXheadinglabel { m }
  { \use:c { CTEXthe#1 } }
\NewExpandableDocumentCommand \CTEXifheadingnumbering { m }
  { \bool_if:cTF { CTEX@#1@numbering } }
\ExplSyntaxOff

\usetheme[max,blue,light]{sjtubeamer}

\makeatletter
\def\sjtubeamer@cover@partnumber{\CTEXheadingnumber{part}}
\def\sjtubeamer@cover@partname{\CTEXheadinglabel{part}}
\def\sjtubeamer@cover@sectionnumber{\CTEXheadingnumber{section}}
\def\sjtubeamer@cover@sectionname{\CTEXheadinglabel{section}}
\def\sjtubeamer@cover@subsectionnumber{\CTEXheadingnumber{subsection}}
\def\sjtubeamer@cover@subsectionname{\CTEXheadinglabel{subsection}}
\makeatother

\ctexset
  {
    part       = { name = {第,部分}, number = \chinese{part} },
    section    = { name = {第,节},   number = \chinese{section} },
    subsection = { name = {第,小节}, number = \chinese{subsection} }
  }

\newcommand\showsectioningpages[1]
  {
    \setbeamertemplate{sectioning pages}[#1]
    \part{部分标题 #1}
    \section{节标题 #1}
    \begin{frame}[plain]\sectionpage\end{frame}
    \subsection{小节标题 #1}
    \begin{frame}[plain]\subsectionpage\end{frame}
  }

\begin{document}
\showsectioningpages{maxplus}
\showsectioningpages{max}
\showsectioningpages{min}
\end{document}
